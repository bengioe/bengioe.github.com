class Squares:
    def __init__(self):
        self.nsquares = 1
        self.size = 5 # size of the observation
        self.side = 2 # size of squares
    @property
    def nactions(self):
        return self.nsquares * 4
    def genRandomSample(self):
        """
        returns (state, action, next state)
        """
        p_0 = pos = [randint(0,self.size-self.side,2) for i in range(self.nsquares)]
        action = randint(0,self.nactions)
        delta = [(1,0),(-1,0),(0,1),(0,-1)]
        s_0 = numpy.zeros([self.size]*2, 'float32')
        for i in range(self.nsquares):
            s_0[pos[i][0]:pos[i][0]+self.side,
                pos[i][1]:pos[i][1]+self.side] = 1
        pos = [p+delta[action%4] if action//4==i else p for i,p in enumerate(pos)]
        pos = [numpy.minimum(numpy.maximum(p,0),self.size-self.side) for p in pos]
        s_1 = numpy.zeros([self.size]*2, 'float32')
        for i in range(self.nsquares):
            s_1[pos[i][0]:pos[i][0]+self.side,
                pos[i][1]:pos[i][1]+self.side] = 0.5
        return (s_0.flatten(), action, s_1.flatten(), p_0)

Model

my model looks like this:

  • input $x$, or state $s_t$
  • an encoder $f(x)\to h$
  • a decoder $g(h) \to r$
  • a policy $\pi_k(x) \to P_k(a|x)$

To train, we minimize

  • autoencoder loss $$ \sum_i (x_i - r_i)^2 $$
  • feature selectivity $$ -\sum_k \log \sum_a P_k(a|s) sel(a,s,s',k) \tag{1}$$

where the policy selectivity $sel(.)$ is $$ sel(a,s_t,s_{t+1},k) = \frac{\|f(s_{t+1})_k - f(s_t)_k\|}{ \sum_{k'} \|f(s_{t+1})_{k'} - f(s_t)_{k'}\|}$$

I actually train for (1) through samples of $s_{t+1}$ and $a_t$.

I train $k$ softmax policies over 4 actions.

def make_fc_mat(i,o):
    k = numpy.sqrt(6./(i+o))
    return numpy.float32(numpy.random.uniform(-k,k,(i,o)))

def make_fc(config, act):
    params = []
    for i in range(len(config)-1):
        params += [make_fc_mat(config[i], config[i+1]), numpy.zeros(config[i+1],'float32')]
    params = map(theano.shared, params)
    def forward(x):
        for i in range(len(config)-1):
            x = act(T.dot(x, params[i*2]) + params[i*2+1])
        return x
    return forward, params

def make_autoencoder(config):
    f, f_params = make_fc(config, T.tanh)
    g, g_params = make_fc(config[::-1], T.tanh)
    autoencode = lambda x: (f(x), g(f(x)))
    return autoencode, f_params + g_params

def make_policy(config):
    return make_fc(config, T.tanh)

def make_model(k, config, nactions):
    
    autoencode, ae_params = make_autoencoder(config + [k])
    pi, pi_params = make_policy(config+[k*nactions])
    
    return autoencode, pi, ae_params + pi_params

def sample_selectivity(a, f, fp):
    #f = theano.printing.Print('f')(f)
    #fp = theano.printing.Print('fp')(fp)
    return abs(f - fp) / (1e-4 + T.sum(abs(f - fp), axis=1)[:, None])

def main():
    k = 6
    nactions = 4
    config = [5*5, 512, 256, 256]
    lr = theano.shared(numpy.array(0.025,'float32'))
    
    environment = Squares()
    
    # create weights, curried theano graph
    autoencode, pi, params = make_model(k, config, nactions)
    
    theano.config.compute_test_value = 'off'
    # theano inputs
    st = T.matrix()
    st.tag.test_value = numpy.zeros((32,5*5),'float32')
    stp1 = T.matrix()
    stp1.tag.test_value = numpy.zeros((32,5*5),'float32')
    at = T.ivector()
    at.tag.test_value = numpy.zeros((32,),'int32')
    
    # the high level
    ft, rt = autoencode(st) # features, reconstruction
    ftp1, _ = autoencode(stp1) # same, but we won't use reconstruction of t+1
    policy = T.nnet.softmax(pi(st).reshape((st.shape[0] * k, nactions))) #P_k(a|s)
    policy = policy.reshape((-1, k, nactions))
    
    reconstruction_loss = T.mean(abs(st-rt))
    
    # since we sample some a_t, this is the selectivity for the particular a_t actions that we sampled:
    selectivity_of_at = policy[T.arange(st.shape[0]), :, at] * sample_selectivity(at, ft, ftp1)
    #print selectivity_of_at.tag.test_value
    selectivity_loss = -T.mean(T.log(selectivity_of_at + 1))
    
    total_loss = reconstruction_loss + 1e-8 * selectivity_loss
    
    
    gradients = T.grad(total_loss, params)
    updates = [(i,i-lr*gi) for i,gi in zip(params, gradients)]
    
    learn_func = theano.function([st,stp1,at], [selectivity_loss, reconstruction_loss], updates=updates)
    encode_func = theano.function([st], ft)
    reconstruct_func = theano.function([st],rt)
    
    all_losses = []
    features = []
    recons = []
    for epoch in range(100):
        losses = train(learn_func, environment, 400)
        #print epoch, map(numpy.mean,zip(*losses))
        all_losses += losses
    
        latent_features, real_features = extract_features(encode_func, environment, 200)
        features.append([latent_features, real_features])
        
        st = environment.genRandomSample()[0]
        rt = reconstruct_func([st])[0]
        recons.append([st,rt])
        
        lr.set_value(numpy.float32(lr.get_value() * 0.95))
    plot_results(all_losses, ['selectivity','reconstruction'], features)
    return features, recons


def train(learn, env, niters):
    mbsize = 64
    losses = []
    for i in range(niters):
        s,a,sp,tf = map(numpy.float32, zip(*[env.genRandomSample() for j in range(mbsize)]))
        losses.append(learn(s,sp,numpy.int32(a)))
        #print losses[-1]
        #break
    return losses

def extract_features(encoder, env, niters):
    mbsize = 1
    latent_features = []
    real_features = []
    for i in range(niters):
        s,a,sp,tf = map(numpy.float32, zip(*[env.genRandomSample() for j in range(mbsize)]))
        real_features.append(tf[0])
        latent_features.append(encoder(s)[0])
    return latent_features, real_features

def plot_results(losses, labels, features):
    pp.clf()
    for label, loss,c in zip(labels, map(numpy.float32, zip(*losses)),'rb'):
        pp.plot(loss, label=label, color=c)
        pp.twinx()
    pp.legend(loc=1)

VIDEO_TAG = """<video controls>
 <source src="data:video/x-m4v;base64,{0}" type="video/mp4">
 Your browser does not support the video tag.
</video>"""

def anim_to_html(anim):
    if not hasattr(anim, '_encoded_video'):
        with NamedTemporaryFile(suffix='.mp4') as f:
            anim.save(f.name, fps=20, extra_args=['-vcodec', 'libx264'])
            video = open(f.name, "rb").read()
        anim._encoded_video = video.encode("base64")
    
    return VIDEO_TAG.format(anim._encoded_video)
